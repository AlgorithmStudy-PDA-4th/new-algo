from collections import deque

def find(a,b):
    cnt=0
    for i in range(len(a)):
        if a[i]==b[i]:
            cnt+=1
    return cnt

def solution(begin, target, words):
    answer = 0
    n=len(begin)
    q=deque()
    visited=[0]*len(words)
    q.append([begin,0])
    
    while q:
        x,cnt=q.popleft()
        print(x,cnt)
        if x==target:
            answer=cnt
            break
        else:
            for i in range(len(words)):
                temp=find(x,words[i])
                if temp>=n-1 and visited[i]==0:
                    visited[i]=1
                    q.append([words[i],cnt+1])
        
    
    return answer
