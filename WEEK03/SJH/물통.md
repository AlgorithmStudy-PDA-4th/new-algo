# 음.. 이건 a와 b로된 bfs를 돌리면 됨

```python


from collections import deque

a,b,c=map(int,input().split())
answer=set()

visited=[[0]*(b+1) for i in range(a+1)]
q=deque()
def fill(x,y):
    if visited[x][y]==0:
        visited[x][y]=1
        q.append([x,y])

q.append([0,0])
visited[0][0]=1
def bfs():
    while q:
        x,y=q.popleft()
        z=c-(x+y)
        if x==0:
            answer.add(z)
    
        #a->b
        water=min(x,b-y)
        fill(x-water,y+water)
    
        #b->a
        water=min(y,a-x)
        fill(x+water,y-water)
    
        #a->c
        water=min(x,c-z)
        fill(x-water,y)
    
        #c->a
        water=min(z,a-x)
        fill(x+water,y)

        #b->c
        water=min(y,c-z)
        fill(x,y-water)
        #c->b
        water=min(z,b-y)
        fill(x,y+water)

bfs()
answer=list(answer)
answer.sort()
print(*answer)

```
