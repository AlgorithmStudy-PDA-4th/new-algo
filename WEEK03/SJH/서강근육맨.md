# 그냥 홀수일때는 정렬된 맨뒤 빼주고 양끝 합 비교 짝수는 그대로 

n=int(input())

a=list(map(int,input().split()))
a.sort()
m=0
if len(a)%2==1:
    m=a[-1]
    a=a[:-1]
    n=len(a)//2
    for i in range(n):
        c,d=a[i],a[-i-1]
        m=max(m,c+d)
else:
    n=len(a)//2
    for i in range(n):
        c,d=a[i],a[-i-1]
        m=max(m,c+d)

print(m)
