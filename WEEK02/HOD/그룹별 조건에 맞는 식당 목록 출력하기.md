```sql
SELECT MEMBER_NAME, REVIEW_TEXT, DATE_FORMAT(r.review_date, '%Y-%m-%d')
FROM REST_REVIEW r
JOIN MEMBER_PROFILE m ON r.MEMBER_ID = m.MEMBER_ID
WHERE r.MEMBER_ID = (
    SELECT MEMBER_ID
    FROM REST_REVIEW
    GROUP BY MEMBER_ID
    ORDER BY COUNT(*) DESC
    LIMIT 1)

ORDER BY r.review_date, r.review_text;
```
