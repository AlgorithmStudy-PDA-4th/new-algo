# 이건 돌아가는 톱니 바퀴를 기준으로 잡고 그 이후 는 어떻게 돌까? 판단 하고,
# 이전에는 어떻게 돌까? 다 판단한 후에,
# 다같이 도는게 정답에 가깝다.

# 정답 (원트로 품)

```python
from collections import deque


def si(q):
    q.appendleft(q.pop())

    return q


def ban_si(q):
    q.append(q.popleft())

    return q


t=[]
for i in range(4):
    tom=deque(list(input()))
    t.append(tom)

n=int(input())

for i in range(n):
    num,bun=map(int,input().split())
    a_bun=bun
    b_bun=bun
    visited=[0]*(4)
    
    visited[num-1]=bun
    for j in range(num-1,len(t)-1):
        if t[j][2]!=t[j+1][6]:
            visited[j+1]=-a_bun
            a_bun=-a_bun 
        else:
            break

    for k in range(num-1,0,-1):
        if t[k][6]!=t[k-1][2]:
            visited[k-1]=-b_bun
            b_bun=-b_bun
        else:
            break
    for i in range(len(visited)):
        if visited[i]==1:
            t[i]=si(t[i])
        if visited[i]==-1:
            t[i]=ban_si(t[i])
            
            
print(int(t[0][0])+int(t[1][0])*2+int(t[2][0])*4+int(t[3][0])*8)
```
